## 变量声明
### var
~~~
for(var i=0;i<10;i++){
	setTimeout(function(){
		// setTimeout 所执行的是相同作用域上的i
		console.log(i)
	},100*i)
}
for(var i=0;i<10;i++){
	(function(i){
		setTimeout(function(){
            // 使用一个立即执行的函数表达式（IIFE）来捕获每次迭代的i值
			console.log(i);
		},100*i)
	})(i)
}
~~~

## 解构
### 解构数组
~~~
let input = [1,2]
let [first,second] = input;
console.log(first,second) // 1 2
// 函数参数解构
function f([first, second]: [number, number]) {
    console.log(first);
    console.log(second);
}
f(input) // 报错  // why
// 数组解构
let [first, ...rest] = [1,23,34,34,234]
console.log(first) // 1
console.log(rest) // [23,34,34,234]
~~~

### 展开操作符
展开操作符与解构相反，它可以将一个数组展开成为另一个数组，或将一个对象展开为另一个对象
~~~
let first = [1,2];
let second = [3,4];
let bothPlus = [0, ...first, ...second, 5];
console.log(bothPlus) //[0, 1, 2, 3, 4, 5]
let defaults = {food: 'spicy', price: '$$', ambiance: 'noisy'};
let search = {food: 'rich', ...defaults}
console.log(search) // {food: 'spicy', price: '$$', ambiance: 'noisy'}
let a = {...defaults, food: 'rich'}
console.log(a) // {food: 'rich', price: '$$', ambiance: 'noisy'}
// 展开对象，会丢掉其方法
class C {
    p=12;
    m(){
    }
}
let c = new C();
let clone = { ...c }
clone.p; // 12
clone.m(); // err
~~~
## 接口
TypeScript的核心原则之一是对值所具有的结构进行类型检查。被称为'鸭式变形法'或'结构性子类型化'。
~~~
// 规定好属性类型
interface LabelledValue {
    label: string;
}
function printLabel(labelledObj: LabelledValue){
    console.log(labelledObj.label) // size 10 Object
}
let myObj = {size: 10, label: 'size 10 Object'};
printLabel(myObj)
// 可选属性(判空)
interface SquareConfig{
    color?: string;
    width?: number;
}
function createSquare(config: SquareConfig): {color: string; area: number}{
    let newSquare = {color: 'white', area: 100};
    if(config.color){
        newSquare.color = config.color;
    }
    if(config.width){
        newSquare.area = Math.pow(config.width,2)
    }
    return newSquare;
}
let mySquare = createSquare({color: 'black'})
let mySquare = createSquare({colour: 'black'}) // err
// 解决方法
// 1
let mySquare = createSquare({color: 'black'} as SquareConfig); // ok
// 2
interface SquareConfig {
    color?: string;
    width?: number;
    [propName: string]: any;
}
// 只读属性
interface Point{
    readonly x: number;
    readonly y: number;
}
let p1: Point = {x: 12, y: 122}
p1.x = 23432 // err
// ReadonlyArray<number>
let a: number[] = [1,23,34,2]
let ro: ReadonlyArray<number> = a;
a = ro as number[]
// readonly VS const
作为变量使用用const，做属性使用用readonly
~~~

### 类类型
~~~

#### 类 静态部分类型 && 实例类型
接口只对实例部分进行检查，constructor存在与类的静态部分，不在接口检查部分
~~~
interface ClockConstructor{
    new (hour: number, minute: number);
}
class Clock implements ClockConstructor{ // err
    currentTime: Date;
    constructor(h: number, m: number){}
}
~~~
改造
~~~
// ClockConstructor为构造器函数所用ClockInterface为实例方法所用
interface ClockConstructor{
    new (hour: number, minute: number): ClockInterface;
}
interface ClockInterface {
    tick();
}
function createClock(ctor: ClockConstructor, hour: number,minute: number): ClockInterface{
    return new ctor(hour, minute)
}
class DigitalClock implements ClockInterface {
    constructor(h: number, m: number){}
    tick(){
        console.log('beep beep')
    }
}
class AnalogClock implements ClockInterface {
    constructor(h: number, m:number){}
    tick(){
        console.log('tick')
    }
}
let digital = createClock(DigitalClock, 12, 17); // createClock(DigitalClock, 12, 17) 会检查DigitalClock是否符合构造函数的签名（有tick()）
let analog = createClock(AnalogClock, 7, 32);
digital.tick() 
analog.tick()
~~~
### 继承接口
~~~
// 一个接口可以继承多个接口，创建一个合成接口
interface Shape{
    color: string;
}
interface PenStroke{
    penWidth: number;
}
interface Square extends Shape,PenStroke{
    sideLength: number;
}
let square = <square>{};
square.color = "blue";
square.sideLength = 10;
square.penWidth = 5.0;
~~~
### 混合类型
~~~
interface Counter{
    (start: number): string;
    interval: number;
    reset: void;
}
function getCounter():Counter{
    let counter = <Counter>function (start: number){};
    counter.interval = 123
    counter.reset = function(){};
    return counter;
}
let c=getCounter();
c(10);
c.reset();
c.interval=5.0;
~~~
